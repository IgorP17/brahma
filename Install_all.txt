Создадим сервис, который будет запускать Minikube при старте системы:

# 1. Создаём скрипт запуска Minikube
cat > ~/start-minikube.sh << 'EOF'
#!/bin/bash
# Ждём пока Docker запустится
sleep 10

# Проверяем, запущен ли уже Minikube
if ! minikube status | grep -q "Running"; then
    echo "🚀 Starting Minikube..."
    minikube start --driver=docker
fi
EOF

chmod +x ~/start-minikube.sh

# 2. Создаём директорию для systemd сервисов
mkdir -p ~/.config/systemd/user/

# 3. Создаём сервис Minikube
cat > ~/.config/systemd/user/minikube.service << 'EOF'
[Unit]
Description=Minikube Kubernetes Cluster
After=docker.service
Requires=docker.service

[Service]
Type=oneshot
ExecStart=/home/igor/start-minikube.sh
RemainAfterExit=yes
ExecStop=/usr/local/bin/minikube stop

[Install]
WantedBy=default.target
EOF

# 4. Включаем сервис
systemctl --user daemon-reload
systemctl --user enable minikube.service
systemctl --user start minikube.service

# 5. Проверяем статус
systemctl --user status minikube.service

echo "✅ Готово! Minikube будет запускаться автоматически при старте системы."

#!/bin/bash
echo "⏳ Waiting for Docker to be ready..."

for i in {1..60}; do
    if docker info &>/dev/null; then
        echo "✅ Docker is ready!"
        break
    fi
    sleep 1
done

# Проверяем, запущен ли уже Minikube
if minikube status | grep -q "Running"; then
    echo "ℹ️  Minikube is already running"
    exit 0
fi

# Запускаем Minikube
echo "🚀 Starting Minikube..."
minikube start --driver=docker
EOF

chmod +x ~/start-minikube.sh

# 2. Обновляем сервис (убираем зависимость от docker.service)
cat > ~/.config/systemd/user/minikube.service << 'EOF'
[Unit]
Description=Minikube Kubernetes Cluster
After=default.target

[Service]
Type=oneshot
ExecStart=/home/igor/start-minikube.sh
RemainAfterExit=yes
ExecStop=/usr/local/bin/minikube stop
TimeoutStartSec=120

[Install]
WantedBy=default.target
EOF

# 3. Перезапускаем сервис
systemctl --user daemon-reload
systemctl --user enable minikube.service

# 4. Тестируем
systemctl --user start minikube.service
