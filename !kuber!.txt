 Решение 1: Автозапуск Minikube при старте системы
sudo tee /etc/systemd/system/minikube.service << 'EOF'
[Unit]
Description=Minikube
After=docker.service
Requires=docker.service

[Service]
Type=oneshot
RemainAfterExit=yes
User=igor
ExecStart=/usr/local/bin/minikube start --driver=docker --wait=false
ExecStop=/usr/local/bin/minikube stop
TimeoutStartSec=300

[Install]
WantedBy=multi-user.target
EOF

Включи сервис
sudo systemctl daemon-reload
sudo systemctl enable minikube.service

проверь
sudo systemctl start minikube.service
minikube status

# 1. Удали все поды (Kubernetes создаст новые с новым образом)
kubectl delete pod -l app=brahma-webui

# 2. Подожди 30-40 секунд
sleep 40

# 3. Проверь, что под пересоздался
kubectl get pods -l app=brahma-webui

# 4. Проверь время создания (должно быть сейчас)
kubectl get pod -l app=brahma-webui -o jsonpath='{.items[0].metadata.creationTimestamp}'

# 5. Проверь содержимое
kubectl exec -it $(kubectl get pod -l app=brahma-webui -o jsonpath='{.items[0].metadata.name}') -- \
  cat /app/templates/index.html | grep "Welcome to Brahma"
