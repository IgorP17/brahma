6. src/main/java/com/example/processor/Terminal.java
package com.example.processor;

import io.quarkus.hibernate.orm.panache.PanacheEntityBase;
import jakarta.persistence.*;

import java.time.LocalDateTime;

@Entity
@Table(schema = "processor", name = "terminals")
public class Terminal extends PanacheEntityBase {

    @Id
    public String id;

    @Column(columnDefinition = "jsonb", nullable = false)
    public String data; // JSONB строка

    @Column(nullable = false)
    public String status = "PENDING";

    @Column(name = "created_at")
    public LocalDateTime createdAt;

    @Column(name = "updated_at")
    public LocalDateTime updatedAt;

    @PrePersist
    protected void onCreate() {
        createdAt = updatedAt = LocalDateTime.now();
    }

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

7. src/main/java/com/example/processor/TerminalRegistrationConsumer.java
package com.example.processor;

import io.smallrye.mutiny.Uni;
import org.eclipse.microprofile.reactive.messaging.Incoming;
import org.eclipse.microprofile.reactive.messaging.Outgoing;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import jakarta.transaction.Transactional;

@ApplicationScoped
public class TerminalRegistrationConsumer {

    @Inject
    TerminalRegisteredProducer kafkaProducer;

    @Incoming("registration-in") // ← имя канала из application.properties
    @Outgoing("registered-out") // ← имя канала для отправки (опционально, можно отправить вручную)
    @Transactional
    public Uni<Void> processRegistration(TerminalRegistrationMessage message) {
        String id = message.id;
        String dataJson = message.dataJson;

        System.out.println("📥 Kafka: received registration for " + id);

        // Проверим, есть ли уже такой терминал
        Terminal existing = Terminal.findById(id);

        if (existing != null) {
            // Обновим статус
            existing.status = "REGISTERED";
            existing.data = dataJson;
            existing.persist();
        } else {
            // Создадим новый
            Terminal terminal = new Terminal();
            terminal.id = id;
            terminal.data = dataJson;
            terminal.status = "REGISTERED";
            terminal.persist();
        }

        // Отправим подтверждение в Kafka
        kafkaProducer.send(id, "REGISTERED");

        System.out.println("✅ DB: terminal " + id + " updated to REGISTERED");
        System.out.println("📤 Kafka: sent confirmation for " + id);

        return Uni.createFrom().voidItem();
    }
}

8. src/main/java/com/example/processor/TerminalRegisteredProducer.java
package com.example.processor;

import org.eclipse.microprofile.reactive.messaging.Emitter;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;

@ApplicationScoped
public class TerminalRegisteredProducer {

    @Inject
    @Channel("registered-out") // ← имя канала из application.properties
    Emitter<TerminalRegisteredMessage> emitter;

    public void send(String id, String status) {
        emitter.send(new TerminalRegisteredMessage(id, status));
    }
}
